A button that scrolls to the top.

## Usage

Add this component to your page.

```tsx title="app/docs/[[...slug]]/page.tsx"
import { RollButton } from 'fumadocs-ui/components/roll-button';

<DocsPage toc={toc}>
  <RollButton />
  <DocsBody>
    <h1>{page.title}</h1>
    <MDX />
  </DocsBody>
</DocsPage>;
```
